Удалить пустые строки и комментарии
sed '/ *#/d; /^ *$/d' - 

Удалить строчки содержащие свап
sed -i '/swap/d' /etc/fstab

Закоментировать строчку
sed -e '/swap/ s/^#*/#/' -i /etc/fstab

Заменить строчку
sed -i 's/archive.ubuntu.com/mirror.yandex.ru/g' /etc/apt/sources.list


a вставить строку после заданной
i вставить после заданной
c замена строк
y замена символов
r вставить данные из файла

sed -n '/test/=' file - вывести номера строк содержащие значение
sed '/word/r file' file - вставить содержимое файла после слова

упорядочить файлы и удалить расширение
ls ~/zim/man/ | sort | sed -e 's/\.txt//' > list
